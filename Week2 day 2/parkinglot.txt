class Vehicle
    attributes: vehicleNumber, entryTime
    method calculateFee(exitTime)

class Car extends Vehicle
    method calculateFee(exitTime)
        hours = ceil(exitTime - entryTime)
        fee = 50 + (hours-1)*30
        return fee

class Bike extends Vehicle
    method calculateFee(exitTime)
        hours = ceil(exitTime - entryTime)
        fee = 20 + (hours-1)*10
        return fee

class Truck extends Vehicle
    method calculateFee(exitTime)
        hours = ceil(exitTime - entryTime)
        fee = 100 + (hours-1)*50
        return fee

main
    car = Car("KA05AB1234",10:00)
    bike = Bike("KA05XY7890",11:15)
    truck = Truck("KA09TR5566",09:30)
    print car.calculateFee(13:30)
    print bike.calculateFee(12:00)
    print truck.calculateFee(14:00)
