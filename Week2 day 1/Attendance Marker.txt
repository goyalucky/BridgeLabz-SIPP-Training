Start

DEFINE students[rows][days]   // rows = number of students, days = 30/31

METHOD markAttendance(studentId, day, status)
    students[studentId][day] = status   // 'P' or 'A'

METHOD updateAttendance(studentId, day, newStatus)
    students[studentId][day] = newStatus

METHOD generateReport(studentId)
    count = 0
    FOR day IN 1..30
        IF students[studentId][day] == 'P'
            count++
    PRINT "Student", studentId, "Present", count, "days"

MAIN
    markAttendance(1,1,'P')
    markAttendance(1,2,'P')
    markAttendance(1,3,'A')
    updateAttendance(1,3,'P')
    generateReport(1)

End
