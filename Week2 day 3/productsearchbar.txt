function binarySearch(catalog, target)
    left = 0
    right = length(catalog)-1
    while left <= right
        mid = (left+right)/2
        if catalog[mid] == target
            return true
        else if catalog[mid] < target
            left = mid+1
        else
            right = mid-1
    return false

main
    catalog = ["AirPods","Backpack","Camera","Laptop","Shoes","Smartphone","Watch"]
    if binarySearch(catalog,"Laptop")
        print "Laptop available"
    else
        print "Not available"
    if binarySearch(catalog,"Headphones")
        print "Headphones available"
    else
        print "Not available"

End
