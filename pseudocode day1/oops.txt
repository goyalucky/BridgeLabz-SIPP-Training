CLASS Pet
    ATTRIBUTES: name, age
    CONSTRUCTOR(name, age) → set values

CLASS Dog EXTENDS Pet
    CONSTRUCTOR(name, age) → super(name, age)

CLASS Cat EXTENDS Pet
    CONSTRUCTOR(name, age) → super(name, age)

CLASS Bird EXTENDS Pet
    CONSTRUCTOR(name, age) → super(name, age)

MAIN:
    CREATE Dog("Buddy", 3)
    CREATE Cat("Mittens", 2)
    CREATE Bird("Tweety", 1)
End
